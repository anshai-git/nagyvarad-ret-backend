1 import { Elysia } from 'elysia';
 2 import AuthRouter from './modules/auth/auth.router';
 3 import DailyVerseRouter from './modules/public/daily-verse/daily-verse.router';
 4 import AdminRouter from './modules/admin/admin.router';
 5 
 6 const authRoutes: Elysia = new Elysia()
 7     .group('/auth',         app => app.use(AuthRouter));
 8 
 9 const adminRoutes: Elysia = new Elysia()
10     .group('/admin',        app => app.use(AdminRouter));
11 
12 const publicRoutes: Elysia = new Elysia()
13     .group('/daily-verse',  app => app.use(DailyVerseRouter));
14 
15 
16 const appRouter: Elysia = new Elysia()
17     .use(authRoutes)
18     .use(publicRoutes)
19     .use(adminRoutes);
20 
21 export default appRouter;
