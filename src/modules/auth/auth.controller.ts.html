1 import { Option, getOrElse, isSome } from 'fp-ts/lib/Option';
 2 import { ApiError } from '../../common/api.error';
 3 import { ApiResponse } from '../../common/api.response';
 4 import { ErrorCodes } from '../../common/constants';
 5 import { LogInResponse } from '../../common/model/response/log-in.response';
 6 import AuthService from './auth.service';
 7 import { Context } from 'elysia';
 8 import { AuthToken } from '../../common/model/auth-token';
 9 import { Credentials } from '../../common/model/credentials';
10 import { LogInRequest } from '../../common/model/request/log-in.request';
11 import { ApiRequest } from '../../common/api.request';
12 import { pipe } from 'fp-ts/lib/function';
13 
14 class AuthController {
15 
16     async logIn(request: ApiRequest<LogInRequest>): Promise<ApiResponse<LogInResponse | null>> {
17         const credentials: Credentials = new Credentials(request.payload.username, request.payload.password);
18         const response: Promise<ApiResponse<LogInResponse | null>> = AuthService.logIn(credentials)
19             .then((token) => {
20                 const authToken: string | null = pipe(
21                     token,
22                     getOrElse(() => null)
23                 )
24                 if(!this.isTokenValid(authToken)) throw new ApiError(ErrorCodes.API_ERROR,'invalid token generated');
25                 const logInResponse: LogInResponse = new LogInResponse(authToken);
26                 return ApiResponse.forSuccess(logInResponse);
27             })
28             .catch((error: Error) => {
29                 if(error instanceof ApiError) {
30                     return ApiResponse.forFailure([error]);
31                 }
32                 return ApiResponse.forFailure([new ApiError('', 'something went wrong')]);
33             })
34 
35         return response;
36     }
37 
38     isTokenValid(token: Option<AuthToken>): boolean {
39         return isSome(token)&&
40             token.value !== null &&
41             token.value.value !== null &&
42             token.value.value !== '';
43     }
44 
45 }
46 
47 const instance = new AuthController();
48 export default instance;
